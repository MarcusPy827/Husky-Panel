33// D-Bus type aliases for the BlueZ ObjectManager interface.
34// InterfaceMap = a{sa{sv}} (interface name → property map)
35// ManagedObjectMap = a{oa{sa{sv}}} (object path → InterfaceMap)
36using BlueZInterfaceMap = QMap<QString, QVariantMap>;
37using BlueZObjectMap = QMap<QDBusObjectPath, BlueZInterfaceMap>;

42namespace panel {
43namespace backend {

45// Monitors and controls Bluetooth via the BlueZ D-Bus API
46// (org.bluez). Exposes adapter and device lists, power state, and
47// connect/disconnect operations. All data is updated reactively via
48// D-Bus signals and marshalled to the Qt main thread.
49class BluetoothInfo : public QObject {
50 Q_OBJECT
51
52 public:
53 // Possible fine-grained states for a device entry exposed to QML.
54 enum class DeviceState {
55 kConnected,
56 kPaired,
57 kConnecting,
58 kFailed,
59 };
60
61 explicit BluetoothInfo(QObject* parent = nullptr);
62 ~BluetoothInfo() override = default;
63
64 // Returns a list of available Bluetooth adapters. Each entry is a
65 // QVariantMap with keys: "address" (QString), "name" (QString),
66 // "path" (QString).
67 QVariantList GetAdapters() const;
68
69 // Returns the D-Bus object path of the currently selected adapter.
70 QString GetSelectedAdapterPath() const;
71
72 // Returns whether the selected adapter is powered on.
73 bool GetEnabled() const;
74
75 // Returns the list of known devices for the selected adapter. Each
76 // entry is a QVariantMap with keys: "address" (QString), "name"
77 // (QString), "icon" (QString — material icon name), "state" (QString
78 // — "connected" | "paired" | "connecting" | "failed").
79 QVariantList GetDevices() const;
80
81 // Powers the selected adapter on or off.
82 void SetEnabled(bool enabled);
83
84 // Changes the active adapter. @p path is a BlueZ D-Bus object path.
85 void SelectAdapter(const QString& path);
86
87 // Initiates an async D-Bus Connect() call on a device. The device's
88 // state transitions to "connecting" immediately, and then to
89 // "connected" or "failed" when the call completes.
90 void ConnectDevice(const QString& address);
91
92 // Initiates an async D-Bus Disconnect() call on a device.
93 void DisconnectDevice(const QString& address);
94
95 signals:
96 void AdaptersChanged();
97 void EnabledChanged();
98 void DevicesChanged();
99
100 private:
101 // Fetches all managed BlueZ objects and rebuilds adapters/devices.
102 void RefreshAll();
103
104 // Rebuilds adapters_ from managed_objects_.
105 void RebuildAdapters();
106
107 // Rebuilds devices_ for the selected adapter from managed_objects_.
108 void RebuildDevices();
109
110 // Returns the material icon name for a BlueZ device icon string.
111 static QString IconForDevice(const QString& bluez_icon);
112
113 // Returns the string key for a DeviceState.
114 static QString StateString(DeviceState s);
115
116 private slots:
117 // Slot: receives org.freedesktop.DBus.ObjectManager.InterfacesAdded.
118 void OnInterfacesAdded(const QDBusObjectPath& path,
119 const BlueZInterfaceMap& ifaces);
120
121 // Slot: receives org.freedesktop.DBus.ObjectManager.InterfacesRemoved.
122 void OnInterfacesRemoved(const QDBusObjectPath& path,
123 const QStringList& ifaces);
124
125 // Slot: receives org.freedesktop.DBus.Properties.PropertiesChanged from
126 // any BlueZ object (wildcard path match). Triggers a full refresh of all
127 // managed objects so that adapter/device data stays consistent.
128 void OnPropertiesChanged(const QString& iface,
129 const QVariantMap& changed, const QStringList& invalidated);
130
131 private:
132 QDBusConnection bus_ = QDBusConnection::systemBus();
133
134 // All objects returned by GetManagedObjects, keyed by object path string.
135 // Stored as QMap<QString, BlueZInterfaceMap> for easy lookup by path string.
136 QMap<QString, BlueZInterfaceMap> managed_objects_;
137
138 QString selected_adapter_path_;
139 QVariantList adapters_;
140 QVariantList devices_;
141
142 // Per-device overrides for connecting/failed states (address → state).
143 QMap<QString, DeviceState> device_state_overrides_;
144};
145
146} // namespace backend
147} // namespace panel
